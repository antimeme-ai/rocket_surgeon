# WU 1.5 — Model Host Skeleton Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Establish the three-binary process topology (daemon → orchestrator → worker) with real IPC from day one, delivering attach/detach lifecycle through to a PyO3-embedded Python skin that loads a nano model.

**Architecture:** Three separate Rust binaries connected by a shared Transport trait over Content-Length framed JSON-RPC on stdio. The daemon spawns an orchestrator on `attach`, the orchestrator spawns a worker, the worker embeds Python via PyO3 and loads/unloads the model through a minimal skin. Same wire format at every boundary.

**Tech Stack:** Rust (edition 2024), PyO3 0.24, tokio 1, clap 4, serde/serde_json, blake3, tracing. Python 3.11+ (PyTorch for skin only).

```rust
use std::io::{self, BufRead, Write};

const MAX_MESSAGE_BYTES: usize = 16 * 1024 * 1024;

#[derive(Debug, thiserror::Error)]
pub enum FramingError {
    #[error("missing Content-Length header")]
    MissingContentLength,
    #[error("invalid Content-Length value")]
    InvalidContentLength,
    #[error("message too large ({0} bytes, max {MAX_MESSAGE_BYTES})")]
    MessageTooLarge(usize),
    #[error("{0}")]
    Io(#[from] io::Error),
}

pub fn read_message(reader: &mut impl BufRead) -> Result<String, FramingError> {
    let mut content_length: Option<usize> = None;

    loop {
        let mut line = String::new();
        let n = reader.read_line(&mut line)?;
        if n == 0 {
            return Err(FramingError::Io(io::Error::new(
                io::ErrorKind::UnexpectedEof,
                "EOF",
            )));
        }

        let trimmed = line.trim_end_matches(['\r', '\n']);

        if trimmed.is_empty() {
            break;
        }

        if let Some((key, value)) = trimmed.split_once(':') {
            if key.eq_ignore_ascii_case("content-length") {
                content_length = Some(
                    value
                        .trim()
                        .parse()
                        .map_err(|_| FramingError::InvalidContentLength)?,
                );
            }
        }
    }

    let content_length = content_length.ok_or(FramingError::MissingContentLength)?;

    if content_length > MAX_MESSAGE_BYTES {
        return Err(FramingError::MessageTooLarge(content_length));
    }

    let mut body = vec![0u8; content_length];
    reader.read_exact(&mut body)?;

    String::from_utf8(body)
        .map_err(|e| FramingError::Io(io::Error::new(io::ErrorKind::InvalidData, e)))
}

pub fn write_message(writer: &mut impl Write, body: &str) -> Result<(), FramingError> {
    let header = format!("Content-Length: {}\r\n\r\n", body.len());
    writer.write_all(header.as_bytes())?;
    writer.write_all(body.as_bytes())?;
    writer.flush()?;
    Ok(())
}
```

```rust
use rocket_surgeon_protocol::jsonrpc::{Request, Response};

use crate::framing::{read_message, write_message};
use crate::{Transport, TransportError};

pub struct StdioTransport<R: BufRead, W: Write> {
    reader: R,
    writer: W,
}

impl<R: BufRead, W: Write> StdioTransport<R, W> {
    pub fn new(reader: R, writer: W) -> Self {
        Self { reader, writer }
    }
}

impl<R: BufRead, W: Write> Transport for StdioTransport<R, W> {
    fn send_request(&mut self, request: &Request) -> Result<(), TransportError> {
        let json = serde_json::to_string(request)?;
        write_message(&mut self.writer, &json)?;
        Ok(())
    }

    fn send_response(&mut self, response: &Response) -> Result<(), TransportError> {
        let json = serde_json::to_string(response)?;
        write_message(&mut self.writer, &json)?;
        Ok(())
    }

    fn recv_request(&mut self) -> Result<Request, TransportError> {
        let raw = read_message(&mut self.reader)?;
        let request: Request = serde_json::from_str(&raw)?;
        Ok(request)
    }

    fn recv_response(&mut self) -> Result<Response, TransportError> {
        let raw = read_message(&mut self.reader)?;
        let response: Response = serde_json::from_str(&raw)?;
        Ok(response)
    }
}
```

```python
"""Minimal Python skin for PyTorch model operations.

No logic, no state management, no IPC. Just the thinnest possible
bridge to PyTorch. Called from Rust worker via PyO3.
"""

from __future__ import annotations

import gc
from typing import Any

import torch
from transformers import AutoConfig, AutoModelForCausalLM

_models: dict[int, torch.nn.Module] = {}
_next_handle: int = 1

_DTYPE_MAP: dict[str, torch.dtype] = {
    "float16": torch.float16,
    "float32": torch.float32,
    "bfloat16": torch.bfloat16,
}


def load_model(source: str, device: str, dtype: str) -> int:
    global _next_handle

    torch_dtype = _DTYPE_MAP.get(dtype, torch.float32)
    model = AutoModelForCausalLM.from_pretrained(
        source,
        torch_dtype=torch_dtype,
        device_map=device if device != "cpu" else None,
    )
    if device == "cpu":
        model = model.to(device)

    handle = _next_handle
    _next_handle += 1
    _models[handle] = model
    return handle


def unload_model(handle: int) -> None:
    model = _models.pop(handle)
    del model
    gc.collect()
    if torch.cuda.is_available():
        torch.cuda.empty_cache()


def model_metadata(handle: int) -> dict[str, Any]:
    model = _models[handle]
    config = model.config

    num_layers = getattr(config, "num_hidden_layers", 0)
    num_heads = getattr(config, "num_attention_heads", 0)
    hidden_dim = getattr(config, "hidden_size", 0)

    module_tree = [name for name, _ in model.named_modules() if name]

    return {
        "num_layers": num_layers,
        "num_heads": num_heads,
        "hidden_dim": hidden_dim,
        "module_tree": module_tree,
    }
```

```rust
use std::process::{Child, Command, Stdio};

use rocket_surgeon_protocol::jsonrpc::{Request, Response};
use rocket_surgeon_transport::framing::{read_message, write_message};
use rocket_surgeon_transport::TransportError;

pub struct WorkerHandle {
    child: Child,
    reader: BufReader<std::process::ChildStdout>,
    writer: std::process::ChildStdin,
}

impl WorkerHandle {
    pub fn spawn(worker_bin: &str, log_level: &str) -> anyhow::Result<Self> {
        let mut child = Command::new(worker_bin)
            .arg("--log-level")
            .arg(log_level)
            .stdin(Stdio::piped())
            .stdout(Stdio::piped())
            .stderr(Stdio::inherit())
            .spawn()?;

        let stdin = child.stdin.take().expect("worker stdin");
        let stdout = child.stdout.take().expect("worker stdout");

        Ok(Self {
            child,
            reader: BufReader::new(stdout),
            writer: stdin,
        })
    }

    pub fn send_request(&mut self, request: &Request) -> Result<(), TransportError> {
        let json = serde_json::to_string(request)?;
        write_message(&mut self.writer, &json)?;
        Ok(())
    }

    pub fn recv_response(&mut self) -> Result<Response, TransportError> {
        let raw = read_message(&mut self.reader)?;
        let response: Response = serde_json::from_str(&raw)?;
        Ok(response)
    }

    pub fn is_alive(&mut self) -> bool {
        matches!(self.child.try_wait(), Ok(None))
    }

    pub fn kill(&mut self) {
        let _ = self.child.kill();
        let _ = self.child.wait();
    }
}

impl Drop for WorkerHandle {
    fn drop(&mut self) {
        self.kill();
    }
}
```

```rust
use rocket_surgeon_protocol::jsonrpc::{Request, Response, RpcError, INTERNAL_ERROR, METHOD_NOT_FOUND};
use rocket_surgeon_protocol::messages::internal;

use crate::worker_handle::WorkerHandle;

pub struct OrchestratorState {
    pub worker: Option<WorkerHandle>,
    pub worker_bin: String,
    pub log_level: String,
}

pub fn dispatch(state: &mut OrchestratorState, request: &Request) -> Response {
    match request.method.as_str() {
        internal::HOST_ATTACH => handle_attach(state, request),
        internal::HOST_DETACH => handle_detach(state, request),
        _ => Response::error(
            request.id.clone(),
            RpcError {
                code: METHOD_NOT_FOUND,
                message: format!("Orchestrator: unknown method '{}'", request.method),
                data: None,
            },
        ),
    }
}

fn handle_attach(state: &mut OrchestratorState, request: &Request) -> Response {
    if state.worker.is_some() {
        return Response::error(
            request.id.clone(),
            RpcError {
                code: INTERNAL_ERROR,
                message: "Worker already running".to_owned(),
                data: None,
            },
        );
    }

    let mut worker = match WorkerHandle::spawn(&state.worker_bin, &state.log_level) {
        Ok(w) => w,
        Err(e) => {
            return Response::error(
                request.id.clone(),
                RpcError {
                    code: INTERNAL_ERROR,
                    message: format!("Failed to spawn worker: {e}"),
                    data: None,
                },
            );
        }
    };

    if let Err(e) = worker.send_request(request) {
        worker.kill();
        return Response::error(
            request.id.clone(),
            RpcError {
                code: INTERNAL_ERROR,
                message: format!("Failed to send to worker: {e}"),
                data: None,
            },
        );
    }

    let response = match worker.recv_response() {
        Ok(r) => r,
        Err(e) => {
            worker.kill();
            return Response::error(
                request.id.clone(),
                RpcError {
                    code: INTERNAL_ERROR,
                    message: format!("Worker failed during attach: {e}"),
                    data: None,
                },
            );
        }
    };

    if response.error.is_none() {
        state.worker = Some(worker);
    } else {
        worker.kill();
    }

    response
}

fn handle_detach(state: &mut OrchestratorState, request: &Request) -> Response {
    let worker = match state.worker.as_mut() {
        Some(w) => w,
        None => {
            return Response::error(
                request.id.clone(),
                RpcError {
                    code: INTERNAL_ERROR,
                    message: "No worker running".to_owned(),
                    data: None,
                },
            );
        }
    };

    if let Err(e) = worker.send_request(request) {
        state.worker = None;
        return Response::error(
            request.id.clone(),
            RpcError {
                code: INTERNAL_ERROR,
                message: format!("Failed to send detach to worker: {e}"),
                data: None,
            },
        );
    }

    let response = match worker.recv_response() {
        Ok(r) => r,
        Err(e) => {
            state.worker = None;
            return Response::error(
                request.id.clone(),
                RpcError {
                    code: INTERNAL_ERROR,
                    message: format!("Worker failed during detach: {e}"),
                    data: None,
                },
            );
        }
    };

    state.worker = None;
    response
}
```

```rust
use rocket_surgeon_protocol::messages::{HostAttachRequest, HostAttachResponse, HostDetachRequest, internal};
use rocket_surgeon_transport::framing::{read_message, write_message};

    reader: BufReader<std::process::ChildStdout>,
    writer: std::process::ChildStdin,
    next_id: i64,
}

impl OrchestratorHandle {
    pub fn spawn(orchestrator_bin: &str, worker_bin: &str, log_level: &str) -> anyhow::Result<Self> {
        let mut child = Command::new(orchestrator_bin)
            .arg("--worker-bin")
            .arg(worker_bin)
            .arg("--log-level")
            .arg(log_level)
            .stdin(Stdio::piped())
            .stdout(Stdio::piped())
            .stderr(Stdio::inherit())
            .spawn()?;

        let stdin = child.stdin.take().expect("orchestrator stdin");
        let stdout = child.stdout.take().expect("orchestrator stdout");

        Ok(Self {
            child,
            reader: BufReader::new(stdout),
            writer: stdin,
            next_id: 1,
        })
    }

    fn next_request_id(&mut self) -> RequestId {
        let id = self.next_id;
        self.next_id += 1;
        RequestId::Number(id)
    }

    pub fn attach(&mut self, req: &HostAttachRequest) -> anyhow::Result<HostAttachResponse> {
        let params = serde_json::to_value(req)?;
        let request = Request::new(self.next_request_id(), internal::HOST_ATTACH, params);

        let json = serde_json::to_string(&request)?;
        write_message(&mut self.writer, &json)?;

        let raw = read_message(&mut self.reader)?;
        let response: Response = serde_json::from_str(&raw)?;

        if let Some(err) = response.error {
            anyhow::bail!("orchestrator attach failed: {}", err.message);
        }

        let result = response.result.ok_or_else(|| anyhow::anyhow!("no result in response"))?;
        let attach_resp: HostAttachResponse = serde_json::from_value(result)?;
        Ok(attach_resp)
    }

    pub fn detach(&mut self, handle: u64) -> anyhow::Result<()> {
        let req = HostDetachRequest { model_handle: handle };
        let params = serde_json::to_value(&req)?;
        let request = Request::new(self.next_request_id(), internal::HOST_DETACH, params);

        let json = serde_json::to_string(&request)?;
        write_message(&mut self.writer, &json)?;

        let raw = read_message(&mut self.reader)?;
        let response: Response = serde_json::from_str(&raw)?;

        if let Some(err) = response.error {
            anyhow::bail!("orchestrator detach failed: {}", err.message);
        }
        Ok(())
    }

    pub fn kill(&mut self) {
        let _ = self.child.kill();
        let _ = self.child.wait();
    }
}
```

```rust
        .or_else(|| find_sibling_binary("rs-worker"))
        .unwrap_or_else(|| "rs-worker".to_owned());

    let mut session = Session::new();
    let mut trace_log = TraceLog::new();
    let mut reader = BufReader::new(io::stdin().lock());
    let mut writer = io::stdout().lock();
    let mut orchestrator: Option<OrchestratorHandle> = None;
    let mut model_handle: Option<u64> = None;

    loop {
        let raw = match read_message(&mut reader) {
            Ok(msg) => msg,
            Err(e) => {
                info!("connection closed: {e}");
                break;
            }
        };

        trace_log.record(Direction::Inbound, &raw);

        let request: rocket_surgeon_protocol::jsonrpc::Request = match serde_json::from_str(&raw) {
            Ok(r) => r,
            Err(e) => {
                let err_resp = serde_json::json!({
                    "jsonrpc": "2.0",
                    "id": null,
                    "error": {
                        "code": rocket_surgeon_protocol::jsonrpc::PARSE_ERROR,
                        "message": format!("Parse error: {e}"),
                    }
                });
                let resp_json = err_resp.to_string();
                trace_log.record(Direction::Outbound, &resp_json);
                if let Err(e) = write_message(&mut writer, &resp_json) {
                    error!("failed to write response: {e}");
                    break;
                }
                continue;
            }
        };

        let response = dispatch(&mut session, &request);
        let resp_json = serde_json::to_string(&response).expect("serialize response");

        if request.method == method::ATTACH && response.error.is_none() {
            if let Some(params) = &request.params {
                if let Ok(attach_req) = serde_json::from_value::<AttachRequest>(params.clone()) {
                    let host_req = HostAttachRequest {
                        model_source: attach_req.model_path.clone(),
                        model_family: attach_req.model_family.clone(),
                        device: attach_req.device.clone(),
                        dtype: attach_req.dtype,
                        rank: 0,
                        config: attach_req.config.clone(),
                    };

                    match OrchestratorHandle::spawn(&orchestrator_bin, &worker_bin, &cli.log_level) {
                        Ok(mut orch) => {
                            match orch.attach(&host_req) {
                                Ok(host_resp) => {
                                    info!(
                                        "model loaded: handle={}, layers={}, heads={}, dim={}, modules={}",
                                        host_resp.model_handle,
                                        host_resp.num_layers,
                                        host_resp.num_heads,
                                        host_resp.hidden_dim,
                                        host_resp.module_tree.len(),
                                    );
                                    model_handle = Some(host_resp.model_handle);
                                    orchestrator = Some(orch);
                                }
                                Err(e) => {
                                    warn!("orchestrator attach failed: {e} — session state already transitioned");
                                }
                            }
                        }
                        Err(e) => {
                            warn!("failed to spawn orchestrator: {e} — session state already transitioned");
                        }
                    }
                }
            }
        }

        if request.method == method::DETACH && response.error.is_none() {
            if let Some(ref mut orch) = orchestrator {
                if let Some(handle) = model_handle.take() {
                    if let Err(e) = orch.detach(handle) {
                        warn!("orchestrator detach failed: {e}");
                    }
                }
            }
            orchestrator = None;
        }

        trace_log.record(Direction::Outbound, &resp_json);

        if let Err(e) = write_message(&mut writer, &resp_json) {
            error!("failed to write response: {e}");
            break;
        }
    }

    drop(orchestrator);

    info!(
        "rocket-surgeon shutting down ({} messages traced)",
        trace_log.len()
    );
}
```

```python
"""End-to-end integration test: daemon → orchestrator → worker → Python skin.

Sends JSON-RPC over the daemon's stdin, verifies the full process chain works.
Requires: cargo build --workspace (all three binaries built).
"""

from __future__ import annotations

import json
import os
import subprocess
import sys
from pathlib import Path


def find_binary(name: str) -> str:
    target_dir = Path(__file__).parent.parent / "target" / "debug"
    candidate = target_dir / name
    if candidate.exists():
        return str(candidate)
    raise FileNotFoundError(f"Binary not found: {candidate}")


def send_message(proc: subprocess.Popen, msg: dict) -> dict:
    body = json.dumps(msg)
    header = f"Content-Length: {len(body)}\r\n\r\n"
    proc.stdin.write(header.encode())
    proc.stdin.write(body.encode())
    proc.stdin.flush()

    # Read response header
    headers = b""
    while b"\r\n\r\n" not in headers:
        byte = proc.stdout.read(1)
        if not byte:
            raise EOFError("daemon closed")
        headers += byte

    header_str = headers.decode()
    content_length = None
    for line in header_str.split("\r\n"):
        if line.lower().startswith("content-length:"):
            content_length = int(line.split(":", 1)[1].strip())
    if content_length is None:
        raise ValueError(f"No Content-Length in response headers: {header_str!r}")

    body_bytes = proc.stdout.read(content_length)
    return json.loads(body_bytes)


def test_full_lifecycle():
    daemon_bin = find_binary("rocket-surgeon")
    orchestrator_bin = find_binary("rs-orchestrator")
    worker_bin = find_binary("rs-worker")

    env = os.environ.copy()
    env["PYTHONPATH"] = str(Path(__file__).parent.parent / "python")

    proc = subprocess.Popen(
        [
            daemon_bin,
            "--orchestrator-bin", orchestrator_bin,
            "--worker-bin", worker_bin,
            "--log-level", "debug",
        ],
        stdin=subprocess.PIPE,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        env=env,
    )

    try:
        # Initialize
        resp = send_message(proc, {
            "jsonrpc": "2.0",
            "id": 1,
            "method": "initialize",
            "params": {
                "client_name": "e2e-test",
                "protocol_version": "0.1.0",
            },
        })
        assert resp.get("error") is None, f"Initialize failed: {resp}"
        state = resp["result"]["state"]
        assert state["status"] == "initialized"

        # Attach
        resp = send_message(proc, {
            "jsonrpc": "2.0",
            "id": 2,
            "method": "attach",
            "params": {
                "model_path": "hf-internal-testing/tiny-random-LlamaForCausalLM",
                "model_family": "llama",
                "device": "cpu",
                "num_ranks": 1,
            },
        })
        assert resp.get("error") is None, f"Attach failed: {resp}"
        state = resp["result"]["state"]
        assert state["status"] == "stopped"
        assert state["model_id"] is not None

        # Detach
        resp = send_message(proc, {
            "jsonrpc": "2.0",
            "id": 3,
            "method": "detach",
        })
        assert resp.get("error") is None, f"Detach failed: {resp}"
        state = resp["result"]["state"]
        assert state["status"] == "initialized"
        assert state["model_id"] is None

        print("PASS: full lifecycle (initialize → attach → detach)")

    finally:
        proc.stdin.close()
        proc.wait(timeout=10)


if __name__ == "__main__":
    test_full_lifecycle()
```
