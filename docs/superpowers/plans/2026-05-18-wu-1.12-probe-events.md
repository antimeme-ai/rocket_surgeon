# WU 1.12: Probe Events Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire the probe lifecycle end-to-end — grammar extension, daemon CRUD, worker evaluation, assert predicates, granularity control — so all 12 TCK scenarios in `probes.feature` go green.

**Architecture:** Daemon owns a `ProbeRegistry` and handles all `rocket/probe` CRUD. After each mutation it propagates full `ProbeDefinition`s to the worker via `_host/update_probes`. The worker caches pre-parsed `ProbePoint`s and evaluates them in the step loop, producing `ProbeFiredEvent`s inline in the step response. The grammar extends from 5 to 6 segments (`model:rank:layer:component:call_index:event`).

**Tech Stack:** Rust, winnow (parser), serde, pyo3 (worker bridge), pytest (E2E)

**Design Spec:** `docs/specs/2026-05-18-probe-events-design.md`

use rocket_surgeon_probes::grammar::ProbePoint;

use crate::adapter::ComponentMap;

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum CaptureMode {
    None,
    Summary,
    Full,
}

pub fn capture_mode(active_probes_count: usize) -> CaptureMode {
    if active_probes_count == 0 {
        CaptureMode::None
    } else {
        CaptureMode::Summary
    }
}

pub fn should_capture(
    map: &ComponentMap,
    module_path: &str,
    call_index: u32,
    active_probes: &[(rocket_surgeon_protocol::types::ProbeDefinition, ProbePoint)],
) -> bool {
    let component = map
        .components
        .iter()
        .find(|c| c.module_path == module_path && c.call_index == call_index);

    let Some(component) = component else {
        return false;
    };

    let Ok(target) = ProbePoint::parse(&component.probe_point) else {
        return false;
    };

    active_probes.iter().any(|(_, pp)| pp.matches(&target))
}

pub fn probe_matches_target(probe_point: &str, target: &str) -> bool {
    let Ok(point) = ProbePoint::parse(probe_point) else {
        return false;
    };
    let Ok(tgt) = ProbePoint::parse(target) else {
        return false;
    };
    point.matches(&tgt)
}

    pub fn evaluate(&self, stats: &TensorStats) -> bool {
        let lhs = self.field.extract(stats);
        match self.op {
            CmpOp::Lt => lhs < self.value,
            CmpOp::Gt => lhs > self.value,
            CmpOp::Le => lhs <= self.value,
            CmpOp::Ge => lhs >= self.value,
            CmpOp::Eq => (lhs - self.value).abs() < f64::EPSILON,
            CmpOp::Ne => (lhs - self.value).abs() >= f64::EPSILON,
        }
    }

impl StatsField {
    fn extract(self, stats: &TensorStats) -> f64 {
        match self {
            Self::Mean => stats.mean,
            Self::Std => stats.std,
            Self::Min => stats.min,
            Self::Max => stats.max,
            Self::AbsMax => stats.abs_max,
            Self::Sparsity => stats.sparsity,
            Self::L2Norm => stats.l2_norm,
        }

// --- Parser ---

fn field_name(input: &mut &str) -> ModalResult<StatsField> {
    let name: String = {
        let first: &str = alpha1.parse_next(input)?;
        let rest: Vec<char> = repeat(
            0..,
            one_of(|c: char| c.is_ascii_alphanumeric() || c == '_'),
        )
        .parse_next(input)?;
        let mut s = first.to_string();
        s.extend(rest);
        s
    };
    match name.as_str() {
        "mean" => Ok(StatsField::Mean),
        "std" => Ok(StatsField::Std),
        "min" => Ok(StatsField::Min),
        "max" => Ok(StatsField::Max),
        "abs_max" => Ok(StatsField::AbsMax),
        "sparsity" => Ok(StatsField::Sparsity),
        "l2_norm" | "norm" => Ok(StatsField::L2Norm),
        _ => Err(winnow::error::ErrMode::Cut(
            winnow::error::ContextError::new(),
        )),
    }
}

fn cmp_op(input: &mut &str) -> ModalResult<CmpOp> {
    alt((
        "<=".map(|_| CmpOp::Le),
        ">=".map(|_| CmpOp::Ge),
        "!=".map(|_| CmpOp::Ne),
        "==".map(|_| CmpOp::Eq),
        "<".map(|_| CmpOp::Lt),
        ">".map(|_| CmpOp::Gt),
    ))
    .parse_next(input)
}

fn float_literal(input: &mut &str) -> ModalResult<f64> {
    let neg: Option<char> = opt('-').parse_next(input)?;
    let int_part: &str = digit1.parse_next(input)?;
    let frac: Option<(&str, &str)> = opt(('.', digit1)).parse_next(input)?;
    let exp: Option<(char, Option<char>, &str)> =
        opt((one_of(['e', 'E']), opt(one_of(['+', '-'])), digit1)).parse_next(input)?;

    let mut s = String::new();
    if neg.is_some() {
        s.push('-');
    }
    s.push_str(int_part);
    if let Some((_, frac_digits)) = frac {
        s.push('.');
        s.push_str(frac_digits);
    }
    if let Some((e, sign, exp_digits)) = exp {
        s.push(e);
        if let Some(sign_char) = sign {
            s.push(sign_char);
        }
        s.push_str(exp_digits);
    }

    s.parse::<f64>()
        .map_err(|_| winnow::error::ErrMode::Cut(winnow::error::ContextError::new()))
}

fn assertion(input: &mut &str) -> ModalResult<Assertion> {
    let field = preceded(space0, field_name).parse_next(input)?;
    let op = preceded(space0, cmp_op).parse_next(input)?;
    let value = preceded(space0, float_literal).parse_next(input)?;
    Ok(Assertion { field, op, value })
}

pub fn handle_probe(
    session: &Session,
    request: &Request,
    registry: &mut ProbeRegistry,
) -> Response {
    if let Err(ref e) = session.require_stopped("rocket/probe") {
        return session_error_to_response(request.id.clone(), e);
    }

    let req: ProbeRequest = match parse_params(request) {
        Ok(r) => r,
        Err(e) => {
            return Response::error(
                request.id.clone(),
                RpcError {
                    code: rocket_surgeon_protocol::jsonrpc::INVALID_PARAMS,
                    message: format!("Invalid params: {e}"),
                    data: None,
                },
            );
        }
    };

    match req {
        ProbeRequest::Define { probe } => {
            match registry.define(*probe) {
                Ok(id) => {
                    let probes = registry.list();
                    let resp = ProbeResponse {
                        probes,
                        probe_id: Some(id),
                    };
                    serialize_envelope(request.id.clone(), session.envelope(resp))
                }
                Err(RegistryError::DuplicateId { id }) => Response::error(
                    request.id.clone(),
                    RpcError::from_error_data(ErrorData::new(
                        ErrorCode::DuplicateProbeId,
                        format!("Probe with id '{id}' already exists"),
                    )),
                ),
                Err(RegistryError::InvalidPoint(e)) => Response::error(
                    request.id.clone(),
                    RpcError::from_error_data(ErrorData::new(
                        ErrorCode::InvalidPoint,
                        format!("Invalid probe point: {e}"),
                    )),
                ),
                Err(e) => Response::error(
                    request.id.clone(),
                    RpcError::from_error_data(ErrorData::new(
                        ErrorCode::InvalidParams,
                        format!("Probe define failed: {e}"),
                    )),
                ),
            }
        }
        ProbeRequest::List {} => {
            let probes = registry.list();
            let resp = ProbeResponse {
                probes,
                probe_id: None,
            };
            serialize_envelope(request.id.clone(), session.envelope(resp))
        }
        ProbeRequest::Enable { probe_id } => match registry.enable(&probe_id) {
            Ok(_) => {
                let probes = registry.list();
                let resp = ProbeResponse {
                    probes,
                    probe_id: Some(probe_id),
                };
                serialize_envelope(request.id.clone(), session.envelope(resp))
            }
            Err(RegistryError::NotFound { id }) => Response::error(
                request.id.clone(),
                RpcError::from_error_data(ErrorData::new(
                    ErrorCode::ProbeNotFound,
                    format!("Probe '{id}' not found"),
                )),
            ),
            Err(e) => Response::error(
                request.id.clone(),
                RpcError::from_error_data(ErrorData::new(
                    ErrorCode::InvalidParams,
                    format!("Enable failed: {e}"),
                )),
            ),
        },
        ProbeRequest::Disable { probe_id } => match registry.disable(&probe_id) {
            Ok(_) => {
                let probes = registry.list();
                let resp = ProbeResponse {
                    probes,
                    probe_id: Some(probe_id),
                };
                serialize_envelope(request.id.clone(), session.envelope(resp))
            }
            Err(RegistryError::NotFound { id }) => Response::error(
                request.id.clone(),
                RpcError::from_error_data(ErrorData::new(
                    ErrorCode::ProbeNotFound,
                    format!("Probe '{id}' not found"),
                )),
            ),

fn resolve_granularity(
    explicit: Option<TickGranularity>,
    scopes: &[GranularityScope],
    current_layer: u32,
) -> Option<TickGranularity> {
    if explicit.is_some() {
        return explicit;
    }
    for scope in scopes {
        if scope_matches(&scope.match_pattern, current_layer) {
            return Some(scope.granularity);
        }
    }
    None
}

fn scope_matches(pattern: &str, layer: u32) -> bool {
    if pattern == "layers[*]" {
        return true;
    }
    if let Some(rest) = pattern.strip_prefix("layers[") {
        if let Some(num_str) = rest.strip_suffix(']') {
            if let Ok(n) = num_str.parse::<u32>() {
                return n == layer;
            }
        }
    }
    false
}
